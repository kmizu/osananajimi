import numpy as np
import matplotlib.pyplot as plt
import matplotlib
matplotlib.rcParams['font.family'] = 'MS Gothic'  # 日本語フォント

# 角度を生成（0から360度）
angles = np.linspace(0, 2 * np.pi, 1000)

# sin, cos, tanの値を計算
sin_values = np.sin(angles)
cos_values = np.cos(angles)

# グラフを描画
plt.figure(figsize=(10, 6))
plt.plot(angles, sin_values, label='sinθ', color='blue')
plt.plot(angles, cos_values, label='cosθ', color='red')
plt.axhline(y=0, color='black', linewidth=0.5)
plt.grid(True, alpha=0.3)
plt.xlabel('角度 (ラジアン)')
plt.ylabel('値')
plt.title('三角関数のグラフ')
plt.legend()
plt.show()
```

# 美月さん向け：花の模様を描くプログラム
import turtle
import math

def draw_flower(petals, radius):
    """petals: 花びらの数、radius: 花びらの半径"""
    angle = 360 / petals
    
    for _ in range(petals):
        # 一つの花びらを描く
        turtle.circle(radius, 60)
        turtle.left(120)
        turtle.circle(radius, 60)
        turtle.left(120)
        
        # 次の花びらの位置へ
        turtle.right(angle)

# 使い方
turtle.speed(0)
turtle.color('pink')
draw_flower(8, 50)
turtle.done()
```
